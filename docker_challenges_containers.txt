1. Basic Image Pull & Run – fetching images, starting containers

docker pull <image_name>

2. Port Mapping & Volume Mounting – connecting host to container, persisting data

docker run --rm -d --name <container_name> -p <host_port>:<container_port> -v /host/data:/container/data <image>:<tag>

3. Multi-Container Networking – communication between multiple containers

docker network create <network_name>
docker run --rm -d --name <container_name1> --network <network_name> -e ENVIRONMENTAL_VARIABLE=value <image>:<tag>
docker run --rm -d --name <container_name2> --network <network_name> -e ENVIRONMENTAL_VARIABLE=value <image>:<tag>

4. Environment & Config Management – passing variables, secrets, configs

docker run --rm -d --name <container_name> -e ENVIRONMENTAL_VARIABLE=value -e ENVIRONMENTAL_VARIABLE=value <image>:<tag>

echo "secret_content" | docker secret create <secret_name> -
docker service create --name <service_name> --secret secret_name <image>:<tag>

docker config create <config_name> <file_path>
docker service create --name <service_name> --config source=<config_name>,target=/etc/nginx/nginx.conf nginx:latest
